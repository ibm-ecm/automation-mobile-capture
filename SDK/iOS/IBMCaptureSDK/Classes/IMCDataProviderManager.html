IMCDataProviderManager

@interface IMCDataProviderManager : NSObject

/// The amount of seconds between every polling call. Defaults to 5 seconds.
@property (readwrite) float pollingInterval;

/// The maximum number of retries for the poll. Defaults to 10 times.
@property (readwrite) int maxRetries;

/// Returns a session that will be local only at the beggining but later on it will be updated with a remote one.
/// @param previousSession If you already have a session, pass it so we don't create a new one
/// @param scenario The scenario for which the session will be created
/// @param propertyContainers The IMCPropertyValueContainer that need to be uploaded to the server
/// @param completion The block that will run when the process is completed. It returns the propertyValueContainer if it's successful or an error if it fails.
- (id<IMCSession> _Nonnull)processSession:(id<IMCSession> _Nullable)previousSession
                              forScenario:(id<IMCScenario> _Nonnull)scenario
                                   atStep:(id<IMCStep> _Nonnull)step
                   withPropertyContainers:(NSArray<id<IMCPropertyValueContainer>> * _Nonnull)propertyContainers
                               completion:(void (^_Nonnull)(NSArray<id<IMCPropertyValueContainer>> * _Nullable, NSError * _Nullable error))completion;

@end
