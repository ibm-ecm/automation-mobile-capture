                              forScenario:(id<IMCScenario> _Nonnull)scenario

                   withPropertyContainers:(NSArray<id<IMCPropertyValueContainer>> * _Nonnull)propertyContainers

    @interface IMCNetworkRequestBuilder : NSObject

/// The URL from where all the enpoints will be called.
@property (nonatomic, readonly, nonnull) NSURL *baseURL;
/// Returns true if the username+password are saved. False otherwise. True doesn't mean that the user is authenticated against the server.
@property (nonatomic, readonly) BOOL authenticated;
/// Called when the session state changes. The new value is what you should store to later on recreate this object with this sessionState.
@property (nonatomic, copy, nullable) void (^ sessionStateChanged)(id<NSObject, NSSecureCoding> _Nonnull sessionState);
/// The IMCNetworkRequester used to perform network calls to the API.
@property (nonatomic, readonly, nonnull) id<IMCNetworkRequester> networkRequester;
/// The IMCNetworkRequester used to upload files. It's different from the regular `networkRequester` because the `fileUploadNetworkRequester` doesn't include any headers
@property (nonatomic, readonly, nonnull) id<IMCNetworkRequester> fileUploadNetworkRequester;

- (instancetype _Nonnull)init NS_UNAVAILABLE;

/// Create a new instance of this object.
/// @param baseURL The base URL from where the enpoints will be called.
/// @param accessId The access Id (if any)
/// @param accessSecretKey The secret access key (if any)
- (instancetype _Nonnull)initWithBaseURL:(NSURL * _Nonnull)baseURL
                                 accessId:(NSString * _Nullable)accessId
                          accessSecretKey:(NSString * _Nullable)accessSecretKey NS_DESIGNATED_INITIALIZER;


/// Create a new instance of this object.
/// @param baseURL The base URL from where the enpoints will be called.
/// @param accessId The access Id (if any)
/// @param accessSecretKey The secret access key (if any)
/// @param sessionState The session state that was given to you on a previous authentication. Use this method to keep the user logged in.
- (instancetype _Nonnull)initWithBaseURL:(NSURL * _Nonnull)baseURL
                                accessId:(NSString * _Nullable)accessId
                         accessSecretKey:(NSString * _Nullable)accessSecretKey
                            sessionState:(id<NSObject, NSSecureCoding> _Nonnull)sessionState;


/// Authenticate agains the server. If the operation succeeds, the sessionState returned should be saved and used to recreate this object later on to avoid having to input again the user credentials.
/// @param credential The username and password combination to login.
/// @param completion The result of the operation. If it succeeds, the sessionState should be stored in Keychain to be retrieved later on. If it fails, the error will contain the reason of the failure.
- (void)authenticateWithCredential:(NSURLCredential * _Nonnull)credential
                        completion:(void (^_Nullable)(id<NSObject, NSSecureCoding> _Nullable sessionState, NSError * _Nullable error))completion;


/// Logout the user. Remove the in memory stored data related to the credentials + call the server to invalidate the token.
- (void)deauthenticate;

/// Invalidates the URLSession used for the IMCNetworkRequester. After calling this, you should not perform any more ntwork calls, if URLSession has been
/// invalidates it asserts as soon as you try to call it.
- (void)invalidate;

@end
