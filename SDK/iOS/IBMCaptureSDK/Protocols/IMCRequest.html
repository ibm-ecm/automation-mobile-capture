IMCRequest

@protocol IMCRequest <NSObject>

/// The entities that the request has. This will be empty until the data has been downloaded from the server and stored locally.
@property (nonatomic, strong, readonly, nonnull) NSArray<id<IMCEntity>> *entities;

/// This block is called as soon as the request finishes.
- (instancetype _Nonnull)completion:(IMCRequestCompletionBlock _Nonnull)completion;
/// This block is called multiple times during the lifetime of the request. The progress is a float ranging from 0 to 1.
- (instancetype _Nonnull)progress:(IMCRequestProgressBlock _Nonnull)progress;
/// This block is called if the request fails. The error returned will contain the cause of the failure.
- (instancetype _Nonnull)error:(IMCRequestErrorBlock _Nonnull)error;
/// This block is called if the request succeeds.
- (instancetype _Nonnull)success:(IMCRequestSuccessBlock _Nonnull)success;

/// This block is called before any change happens.
- (instancetype _Nonnull)willChange:(IMCRequestWillChangeBlock _Nonnull)willChange;
/// This block is called when an IMCEntity is inserted. The block will contain the inserted IMCEntity and the index where's been inserted to.
- (instancetype _Nonnull)inserted:(IMCRequestInsertedBlock _Nonnull)inserted;
/// This block is called when an IMCEntity is deleted. The block will contain the deleted IMCEntity and the index where's been deleted from.
- (instancetype _Nonnull)deleted:(IMCRequestDeletedBlock _Nonnull)deleted;
/// This block is called when an IMCEntity is moved. The block will contain the moved IMCEntity and the indexes where's been moved from and to.
- (instancetype _Nonnull)moved:(IMCRequestMovedBlock _Nonnull)moved;
/// This block is called when an IMCEntity is updated. The block will contain the updated IMCEntity and the index where's been updated.
- (instancetype _Nonnull)updated:(IMCRequestUpdatedBlock _Nonnull)updated;
/// This block is called after all the changes have happened.
- (instancetype _Nonnull)didChange:(IMCRequestDidChangeBlock _Nonnull)didChange;

/// This method triggers the request. Usually, this means setting up an NSFetchedResultsController in order to notify when there are changes in the model
/// and trigger a network request to update the data.
- (void)execute;

/// Suspend any network request that's ongoing.
- (void)suspend;

- (void)startReportingChanges;

- (instancetype _Nonnull)duplicate;

@end
